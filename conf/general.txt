class game:
    # The game ends in failure after this number of rounds has passed without
    # the final dungeon being cleared.
    doom = 30

class dungeon_base:
    # These stats are added to every dungeon's individual stats.
    danger_start  = 1
    danger_step   = 1
    mystery_start = 1
    mystery_step  = 1

class dungeon_sizes:
    # The number of levels in a dungeon of each size.
    small  = 2
    medium = 5
    large  = 8
    huge   = 11

class hero:
    # The starting values of each stat.
    fight_start = 1
    fight_step  = 3
    
    # The number of experience points needed to increment each stat.
    lore_start  = 1
    lore_step   = 3
